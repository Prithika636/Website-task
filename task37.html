// Node.js Backend - Single File const http = require("http"); const url = require("url"); const PORT = 3000; // Temporary database let users = [ { id: 1, name: "Admin", email: "admin@gmail.com" } ]; // Server const server = http.createServer((req, res) => { const parsedUrl = url.parse(req.url, true); // Allow JSON response res.setHeader( "Content-Type", "application/json" ); // Home API if(req.url === "/" && req.method === "GET"){ res.writeHead(200); res.end(JSON.stringify({ message:"Node.js Backend Running" })); } // GET ALL USERS else if( parsedUrl.pathname === "/users" && req.method === "GET" ){ res.writeHead(200); res.end( JSON.stringify(users) ); } // CREATE USER else if( parsedUrl.pathname === "/users" && req.method === "POST" ){ let body = ""; req.on("data",(chunk)=>{ body += chunk; }); req.on("end",()=>{ const data = JSON.parse(body); const newUser = { id: Date.now(), name:data.name, email:data.email }; users.push(newUser); res.writeHead(201); res.end(JSON.stringify({ message:"User Created", user:newUser })); }); } // UPDATE USER else if( parsedUrl.pathname.startsWith("/users/") && req.method === "PUT" ){ const id = Number( parsedUrl.pathname.split("/")[2] ); let body=""; req.on("data",chunk=>{ body += chunk; }); req.on("end",()=>{ const data = JSON.parse(body); const user = users.find( u=>u.id===id ); if(user){ user.name = data.name || user.name; user.email = data.email || user.email; res.end(JSON.stringify({ message:"User Updated", user })); } else{ res.writeHead(404); res.end(JSON.stringify({ message:"User Not Found" })); } }); } // DELETE USER else if( parsedUrl.pathname.startsWith("/users/") && req.method === "DELETE" ){ const id = Number( parsedUrl.pathname.split("/")[2] ); users = users.filter( user=>user.id!==id ); res.end(JSON.stringify({ message:"User Deleted" })); } // ERROR else{ res.writeHead(404); res.end(JSON.stringify({ message:"Route Not Found" })); } }); server.listen(PORT,()=>{ console.log( "Server running at http://localhost:3000" ); });
