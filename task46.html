const jwt = require("jsonwebtoken"); const JWT_SECRET = "your-secret-key"; function authMiddleware(req, res, next) { try { // Get token from Authorization header const authHeader = req.headers.authorization; if (!authHeader || !authHeader.startsWith("Bearer ")) { return res.status(401).json({ success: false, message: "Access token required" }); } const token = authHeader.split(" ")[1]; // Verify token const decoded = jwt.verify(token, JWT_SECRET); // Store user info in request req.user = decoded; next(); } catch (error) { return res.status(403).json({ success: false, message: "Invalid or Expired Token" }); } } module.exports = authMiddleware;
